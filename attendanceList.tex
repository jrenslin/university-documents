% A table for listing names and many empty boxes to check - one per week
% Created for courses at Goethe U. Frankfurt,
% where a regular semester usually has no more than 16 or 17 weeks at max.
%
%---------------------------------------------------------------------------
% Set document class
%------------
\documentclass{article}

%------------
% Encoding
%-----
\usepackage[utf8]{inputenc}

%------------
% Styles
\usepackage[a4paper,margin=2cm,landscape]{geometry}
\pagestyle{empty}

%------------
% Required components
%-----
\usepackage{tabularx}
\usepackage{caption}
\usepackage{array}

%---------------------------------------------------------------------------
% Set variables
%------------

\def \courseMembersTitle {Attendance List}
\def \courseMembersCourseTitle {Tutorium Transnationale Migration in SÃ¼dostasien}
\def \courseMembersTimePeriod {WiSe 2017-18}

%------------
% Counters to determine table size
%-----

\newcounter{row}
\newcounter{max}

\setcounter{row}{1}
\setcounter{max}{19} % Number of the last row + 1 (the condition is *smaller than*: 14 lines means \setcounter{max}{15}

%---------------------------------------------------------------------------
% More
% Thanks: https://tex.stackexchange.com/questions/224315/loop-in-tabular-that-uses-the-count-to-access-current-line-number
%------------

\newtoks\tablines
\newcommand{\addtomytablines}[1]{\tablines\expandafter{\the\tablines#1}}
\loop
  \expandafter\addtomytablines\expandafter{\number\arabic{row} & & & & & & & & & & & & & & & & & \\ \hline}
  \stepcounter{row}
\ifnum \value{row}<\value{max}
\repeat
